Crie uma tabela de 'Generos' com a seguinte estrutura:

id (chave primária, inteiro)

nome_genero (texto: 10 caracteres)

CREATE TABLE generos ( id INT NOT NULL AUTO_INCREMENT, nome_genero VARCHAR(10), PRIMARY KEY (id));
